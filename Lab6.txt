#include <math.h>
#include <stdio.h>
#include <conio.h>
#include <locale>
int main()
{
	setlocale(LC_ALL, "RUS");
	int A[20], min;
	printf("Часть А \n");
	for (int i = 0; i < 20; i++) {
		*(A + i) = rand();
		printf("%d ", *(A + i));
		if ((i + 1) % 5 == 0) {
			printf("\n");
		};
	};
	min = *A;
	for (int i = 1; i < 20; i++) {
		if (min > *(A + i)) {
			min = *(A + i);
		};
	};
	printf("Минимальное = %d \n", min);
	int B[3][5], sov = 0;
	int* nB = &B[0][0];
	printf("Часть В \n");
		for (int i = 0; i < 3; i++) {
			for (int ii = 0; ii < 5; ii++) {
				*(*(B + i) + ii) = rand();
				printf("%d ", *(*(B + i) + ii));
				if (ii == 4) {
					printf("\n");
				}
			}
		}
	for (int i = 0; i < 15; i++) {
		if (*(nB + i) == *(nB + i - 1)) {
			sov++;
		}
	}
	printf("Количество совпадений = %d", sov);
	getchar();
